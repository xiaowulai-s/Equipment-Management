## 1. 串口枚举

### 基础用法

```python
from core.utils.serial_utils import list_serial_ports, get_serial_ports_info

# 简单枚举 - 获取串口名称列表
ports = list_serial_ports()
print(ports)  # ['COM1', 'COM3', 'COM5', ...]

# 详细信息 - 获取串口完整信息
ports_info = get_serial_ports_info()
for info in ports_info:
    print(f"{info.device}: {info.description}")
    # 输出示例:
    # COM1: Intel(R) Active Management Technology - SOL
    # COM3: Prolific USB-to-Serial Comm Port
    # COM5: USB Serial Device
```

### 高级用法

```python
from core.utils.serial_utils import (
    get_serial_port_by_description,
    get_serial_port_details,
    list_serial_ports_as_dict,
    get_port_count,
    is_serial_available
)

# 检查 pyserial 是否可用
if is_serial_available():
    print("串口功能可用")
else:
    print("请先安装 pyserial: pip install pyserial")

# 获取串口数量
count = get_port_count()
print(f"可用串口数量：{count}")

# 根据描述查找串口（支持模糊搜索）
port = get_serial_port_by_description("USB")
if port:
    print(f"找到 USB 串口：{port}")

# 获取指定串口的详细信息
port_info = get_serial_port_details("COM3")
if port_info:
    print(f"设备：{port_info.device}")
    print(f"描述：{port_info.description}")
    print(f"硬件 ID: {port_info.hwid}")
    print(f"厂商 ID: {port_info.vid}")
    print(f"产品 ID: {port_info.pid}")
    print(f"序列号：{port_info.serial_number}")
    print(f"位置：{port_info.location}")
    print(f"制造商：{port_info.manufacturer}")

# 以字典列表形式返回（适合 JSON 序列化）
ports_dict = list_serial_ports_as_dict()
import json
print(json.dumps(ports_dict, indent=2))
```

### 在添加设备对话框中使用

```python
from core.utils.serial_utils import get_serial_ports_info

def populate_serial_port_combo(self, combo_box):
    """填充串口选择下拉框"""
    combo_box.clear()

    ports_info = get_serial_ports_info()
    for info in ports_info:
        # 显示格式：COM3 - Prolific USB-to-Serial Comm Port
        display_text = f"{info.device} - {info.description}"
        combo_box.addItem(display_text, userData=info.device)

    # 获取用户选择的串口
    selected_port = combo_box.currentData()
    print(f"用户选择：{selected_port}")
```

### 常见验证错误

```python
# 错误 1: 缺少必需字段
invalid_config_1 = {

# 错误 2: 不支持的协议类型

    "protocol_type": "modbus_udp",  # 不支持

valid, msg = Device.validate_config(invalid_config_2)
# 输出：不支持的协议类型：modbus_udp

invalid_config_3 = {
    "device_id": "device_001",
    "name": "测试设备",
    "device_type": "sensor",
    "protocol_type": "modbus_tcp",
    "host": "192.168.1.100",
    "port": 70000,  # 超出 1-65535 范围
}
valid, msg = Device.validate_config(invalid_config_3)
# 输出：端口号必须在 1-65535 范围内

# 错误 4: Unit ID 超出范围
invalid_config_4 = {
    "device_id": "device_001",
    "name": "测试设备",
    "device_type": "sensor",
    "protocol_type": "modbus_tcp",
    "host": "192.168.1.100",
    "port": 502,
```

### 在添加设备对话框中使用

```python
from core.device.device_model import Device

def on_add_device(self):
    """处理添加设备"""
    config = {
        "device_id": self.device_id_edit.text(),
        "name": self.name_edit.text(),

        return

    # 配置有效，继续添加设备
    self._device_manager.add_device(config)
    QMessageBox.information(self, "成功", "设备添加成功")
```

## 3. 批量读取寄存器

### 基础用法

```python
from core.protocols.modbus_protocol import ModbusProtocol

# 假设 protocol 已经初始化并连接
protocol = ModbusProtocol(driver=driver, mode="TCP")

# 批量读取多个连续寄存器块
addresses = [
    (0, 10),    # 读取地址 0-9
    (100, 5),   # 读取地址 100-104
    (200, 20)   # 读取地址 200-219
]

results = protocol.read_registers_batch(addresses)
if results:
    for start_addr, values in results.items():
        print(f"地址 {start_addr}: {values}")
        # 输出示例:
        # 地址 0: [100, 200, 150, 180, 220, 190, 210, 230, 240, 250]
        # 地址 100: [50, 60, 70, 80, 90]
        # 地址 200: [10, 20, 30, ..., 200]  # 20 个值
else:
    print("批量读取失败")
```

### 性能对比

```python
import time

# 方式 1: 逐个读取（慢）
start = time.time()
for addr in range(0, 100):
    data = protocol.read_registers(addr, 1)
end = time.time()
print(f"逐个读取耗时：{end - start:.3f}秒")

# 方式 2: 批量读取（快）
start = time.time()
addresses = [(0, 100)]  # 一次性读取 100 个
results = protocol.read_registers_batch(addresses)
end = time.time()
print(f"批量读取耗时：{end - start:.3f}秒")

# 性能提升：批量读取通常快 5-10 倍
```

### 实际应用场景

```python
# 场景 1: 读取多相电力参数
power_addresses = [
    (0, 3),    # A/B/C相电压
    (10, 3),   # A/B/C相电流
    (20, 3),   # A/B/C相功率
    (30, 1),   # 总功率
    (40, 1),   # 功率因数
    (50, 1),   # 频率
]

power_data = protocol.read_registers_batch(power_addresses)

if power_data:
    voltages = power_data.get(0, [])
    currents = power_data.get(10, [])
    powers = power_data.get(20, [])

    print(f"A 相电压：{voltages[0]}V")
    print(f"A 相电流：{currents[0]}A")
    print(f"A 相功率：{powers[0]}W")

# 场景 2: 读取温湿度传感器阵列
sensor_addresses = []
for i in range(10):  # 10 个传感器
    base_addr = i * 10
    sensor_addresses.append((base_addr, 2))  # 每个传感器 2 个寄存器（温度 + 湿度）

sensor_data = protocol.read_registers_batch(sensor_addresses)

for i, (start_addr, values) in enumerate(sensor_data.items()):
    temp = values[0] / 10.0  # 温度带 1 位小数
    humidity = values[1] / 10.0  # 湿度带 1 位小数
    print(f"传感器{i+1}: 温度{temp}°C, 湿度{humidity}%RH")
```

### 错误处理

```python
results = protocol.read_registers_batch(addresses)

if results is None:
    print("批量读取失败：设备未连接或通信错误")
else:
    # 检查哪些地址块读取成功
    success_count = len(results)
    total_count = len(addresses)

    if success_count < total_count:
        print(f"部分读取失败：成功{success_count}/{total_count}")

        # 找出失败的地址块
        for start_addr, count in addresses:
            if start_addr not in results:
                print(f"地址块 {start_addr}-{start_addr + count - 1} 读取失败")

    # 处理成功读取的数据
    for start_addr, values in results.items():
        print(f"地址 {start_addr}: {values}")
```

## 4. pre-commit 使用

### 安装和配置

```bash
# 1. 安装依赖（已安装）
pip install pre-commit black isort flake8

# 2. 安装 pre-commit 钩子（已安装）
pre-commit install

# 3. 验证安装
pre-commit --version
```

### 首次运行

```bash
# 对所有文件运行检查
pre-commit run --all-files

# 输出示例:
# black....................................Passed
# isort....................................Passed
# flake8...................................Passed
# trailing-whitespace......................Passed
# end-of-file-fixer........................Passed
```

### 日常使用

```bash
# 提交代码时自动运行
git add .
git commit -m "feat: 添加新功能"

# pre-commit 会自动检查：
# 1. Black 格式化
# 2. isort 导入排序
# 3. flake8 代码检查
# 4. 空白行检查
# 5. 文件结尾检查

# 如果检查失败，commit 会被阻止
# 修复问题后重新提交
```

### 跳过检查（不推荐）

```bash
# 紧急情况跳过 pre-commit
git commit -m "fix: 紧急修复" --no-verify

# 仅用于紧急情况，常规开发不建议使用
```

### 配置说明

`.pre-commit-config.yaml` 配置：

```yaml
repos:
  # 基础检查
  - repo: https://github.com/pre-commit/pre-commit-hooks
    rev: v4.5.0
    hooks:
      - id: trailing-whitespace    # 删除行尾空白
      - id: end-of-file-fixer      # 确保文件以换行结尾
      - id: check-yaml             # 检查 YAML 语法
      - id: check-json             # 检查 JSON 语法
      - id: check-added-large-files # 检查大文件
      - id: check-merge-conflict   # 检查合并冲突标记
      - id: detect-private-key     # 检测私钥

  # Black 格式化
  - repo: https://github.com/psf/black
    rev: 24.3.0
    hooks:
      - id: black
        language_version: python3
        args: [--line-length=120]

  # isort 导入排序
  - repo: https://github.com/pycqa/isort
    rev: 5.13.2
    hooks:
      - id: isort
        args: [--profile=black, --line-length=120]

  # flake8 代码检查
  - repo: https://github.com/pycqa/flake8
    rev: 7.0.0
    hooks:
      - id: flake8
        args: [--max-line-length=120, --extend-ignore=E203,W503]
```

## 6. 主题扩展

### 深色主题

```python
from ui.styles import AppStyles

class ThemeManager:
    """主题管理器"""

    @staticmethod
    def get_dark_theme() -> str:
        """获取深色主题"""
        return """
            QMainWindow {
                background-color: #0D1117;
                color: #C9D1D9;
            }

            QWidget {
                background-color: #161B22;
                color: #C9D1D9;
            }

            QPushButton {
                background-color: #21262D;
                color: #C9D1D9;
                border: 1px solid #30363D;
                border-radius: 6px;
                padding: 8px 16px;
            }

            QPushButton:hover {
                background-color: #30363D;
                border-color: #58A6FF;
            }

            QLineEdit, QComboBox {
                background-color: #0D1117;
                color: #C9D1D9;
                border: 1px solid #30363D;
                border-radius: 6px;
                padding: 8px 12px;
            }

            QLineEdit:focus, QComboBox:focus {
                border-color: #58A6FF;
            }

            QTreeWidget, QTableWidget {
                background-color: #161B22;
                color: #C9D1D9;
                border: 1px solid #30363D;
            }

            QHeaderView::section {
                background-color: #21262D;
                color: #8B949E;
                border: none;
                padding: 8px;
            }
        """

    @staticmethod
    def apply_theme(theme: str):
        """应用主题"""
        from PySide6.QtWidgets import QApplication
        app = QApplication.instance()
        app.setStyleSheet(theme)

# 使用
# 切换到深色主题
ThemeManager.apply_theme(ThemeManager.get_dark_theme())

# 切换回浅色主题
ThemeManager.apply_theme(AppStyles.MAIN_WINDOW)
```

### 主题切换按钮

```python
class ThemeToggleButton(QPushButton):
    """主题切换按钮"""

    def __init__(self, parent=None):
        super().__init__("🌙 深色模式", parent)
        self._is_dark = False
        self.clicked.connect(self.toggle_theme)

    def toggle_theme(self):
        """切换主题"""
        if self._is_dark:
            ThemeManager.apply_theme(AppStyles.MAIN_WINDOW)
            self.setText("🌙 深色模式")
        else:
            ThemeManager.apply_theme(ThemeManager.get_dark_theme())
            self.setText("☀️ 浅色模式")

        self._is_dark = not self._is_dark
```
