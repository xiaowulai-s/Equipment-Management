## 二、支持的工业通信协议

### 2.1 主要协议列表

| 协议名称 | 协议类型 | 传输层 | 适用场景 | 特点 |
|----------|----------|--------|----------|------|
| Modbus RTU | 串行协议 | RS-232/485 | 短距离、低速设备 | 简单、可靠、广泛应用 |
| Modbus TCP | 以太网协议 | TCP/IP | 网络连接设备 | 传输速度快、距离远 |
| OPC UA | 统一架构协议 | TCP/IP | 异构系统集成 | 安全、可靠、跨平台 |
| Profinet | 实时以太网协议 | TCP/IP | 高性能自动化系统 | 实时性高、确定性强 |
| EtherNet/IP | 工业以太网协议 | TCP/IP | 罗克韦尔设备 | 灵活、可扩展 |
| MQTT | 消息队列协议 | TCP/IP | 物联网设备 | 轻量级、低带宽 |

### 2.2 协议选择策略

| 设备类型 | 推荐协议 | 备选协议 | 选择理由 |
|----------|----------|----------|----------|
| PLC控制器 | OPC UA | Modbus TCP | 支持复杂数据结构，安全可靠 |
| 传感器 | Modbus RTU/TCP | MQTT | 简单易用，成本低 |
| 仪表设备 | Modbus RTU | HART | 广泛应用于仪表行业 |
| 执行器 | Profinet | EtherNet/IP | 实时性要求高，控制精度高 |
| 物联网设备 | MQTT | OPC UA | 低带宽，电池供电 |

## 四、协议驱动架构

### 4.1 驱动接口规范

#### 4.1.1 驱动基类

```csharp
public abstract class ProtocolDriverBase : IDeviceConnection, IDataCollector, IDataWriter, IDeviceStatus, ICommunicationConfig
{
    // 设备信息
    protected DeviceInfo DeviceInfo { get; set; }
    
    // 通信配置
    protected CommunicationSettings Settings { get; set; }
    
    // 驱动状态
    protected DriverState DriverState { get; set; }
    
    // 构造函数
    protected ProtocolDriverBase(DeviceInfo deviceInfo, CommunicationSettings settings)
    {
        DeviceInfo = deviceInfo;
        Settings = settings;
        DriverState = DriverState.Initialized;
    }
    
    // 抽象方法（由具体驱动实现）
    protected abstract Task<bool> DoConnectAsync(CancellationToken cancellationToken);
    protected abstract Task<bool> DoDisconnectAsync();
    protected abstract Task<DeviceData> DoReadDataPointAsync(DataPoint dataPoint, CancellationToken cancellationToken);
    protected abstract Task<WriteResult> DoWriteDataPointAsync(DataPoint dataPoint, object value, CancellationToken cancellationToken);
    protected abstract Task<DeviceStatusInfo> DoGetDeviceStatusAsync(CancellationToken cancellationToken);
    
    // 实现IDeviceConnection接口
    public virtual async Task<bool> ConnectAsync(CancellationToken cancellationToken = default)
    {
        // 连接逻辑模板
        try
        {
            UpdateState(ConnectionState.Connecting);
            var result = await DoConnectAsync(cancellationToken);
            UpdateState(result ? ConnectionState.Connected : ConnectionState.Fault);
            return result;
        }
        catch (Exception ex)
        {
            UpdateState(ConnectionState.Fault);
            // 记录异常
            return false;
        }
    }
    
    // 其他接口方法实现...
}
```

#### 4.1.2 驱动管理器

```csharp
public class DriverManager
{
    private readonly Dictionary<string, ProtocolDriverBase> _drivers;
    private readonly IServiceProvider _serviceProvider;
    private readonly ILogger _logger;
    
    public DriverManager(IServiceProvider serviceProvider, ILogger logger)
    {
        _drivers = new Dictionary<string, ProtocolDriverBase>();
        _serviceProvider = serviceProvider;
        _logger = logger;
    }
    
    // 创建驱动实例
    public ProtocolDriverBase CreateDriver(DeviceInfo deviceInfo, CommunicationSettings settings)
    {
        var driverType = GetDriverType(settings.ProtocolType);
        if (driverType == null)
        {
            throw new NotSupportedException($"Protocol {settings.ProtocolType} is not supported");
        }
        
        var driver = (ProtocolDriverBase)ActivatorUtilities.CreateInstance(_serviceProvider, driverType, deviceInfo, settings);
        _drivers[deviceInfo.DeviceId] = driver;
        
        return driver;
    }
    
    // 获取驱动实例
    public ProtocolDriverBase GetDriver(string deviceId)
    {
        return _drivers.TryGetValue(deviceId, out var driver) ? driver : null;
    }
    
    // 销毁驱动实例
    public bool DestroyDriver(string deviceId)
    {
        if (_drivers.TryGetValue(deviceId, out var driver))
        {
            driver.DisconnectAsync().Wait();
            _drivers.Remove(deviceId);
            return true;
        }
        return false;
    }
    
    // 获取支持的协议列表
    public IEnumerable<string> GetSupportedProtocols()
    {
        // 返回支持的协议列表
        return new[] { "ModbusRTU", "ModbusTCP", "OPCUA", "Profinet", "EthernetIP" };
    }
    
    // 获取驱动类型
    private Type GetDriverType(string protocolType)
    {
        // 根据协议类型返回对应的驱动类型
        switch (protocolType.ToUpper())
        {
            case "MODBUSRTU":
                return typeof(ModbusRtuDriver);
            case "MODBUSTCP":
                return typeof(ModbusTcpDriver);
            case "OPCUA":
                return typeof(OpcUaDriver);
            case "PROFINET":
                return typeof(ProfinetDriver);
            case "ETHERNETIP":
                return typeof(EthernetIpDriver);
            default:
                return null;
        }
    }
}
```

### 4.2 具体驱动实现示例

#### 4.2.1 Modbus TCP驱动实现

```csharp
public class ModbusTcpDriver : ProtocolDriverBase
{
    private ModbusIpMaster _modbusMaster;
    private TcpClient _tcpClient;
    
    public ModbusTcpDriver(DeviceInfo deviceInfo, CommunicationSettings settings) : base(deviceInfo, settings)
    {
    }
    
    protected override async Task<bool> DoConnectAsync(CancellationToken cancellationToken)
    {
        try
        {
            _tcpClient = new TcpClient();
            await _tcpClient.ConnectAsync(Settings.Address, Settings.Port, cancellationToken);
            
            var factory = new ModbusFactory();
            _modbusMaster = factory.CreateIpMaster(_tcpClient);
            _modbusMaster.Transport.ReadTimeout = Settings.Timeout;
            _modbusMaster.Transport.WriteTimeout = Settings.Timeout;
            
            return true;
        }
        catch (Exception ex)
        {
            // 记录异常
            return false;
        }
    }
    
    protected override async Task<bool> DoDisconnectAsync()
    {
        try
        {
            _tcpClient?.Close();
            _tcpClient?.Dispose();
            _modbusMaster = null;
            return true;
        }
        catch (Exception ex)
        {
            // 记录异常
            return false;
        }
    }
    
    protected override async Task<DeviceData> DoReadDataPointAsync(DataPoint dataPoint, CancellationToken cancellationToken)
    {
        try
        {
            object value = null;
            var address = int.Parse(dataPoint.Address);
            var slaveId = DeviceInfo.SlaveId ?? 1;
            
            switch (dataPoint.DataType)
            {
                case DataType.Bool:
                    var coils = await _modbusMaster.ReadCoilsAsync(slaveId, address, 1, cancellationToken);
                    value = coils[0];
                    break;
                case DataType.Short:
                    var registers = await _modbusMaster.ReadHoldingRegistersAsync(slaveId, address, 1, cancellationToken);
                    value = dataPoint.IsSigned ? (short)registers[0] : (ushort)registers[0];
                    break;
                case DataType.Float:
                    registers = await _modbusMaster.ReadHoldingRegistersAsync(slaveId, address, 2, cancellationToken);
                    value = ModbusUtility.GetSingle(registers);
                    break;
                // 其他数据类型处理...
            }
            
            // 应用缩放和偏移
            var processedValue = ApplyScaleAndOffset(value, dataPoint.Scale, dataPoint.Offset);
            
            return new DeviceData
            {
                DataId = Guid.NewGuid().ToString(),
                PointId = dataPoint.PointId,
                DeviceId = DeviceInfo.DeviceId,
                ParameterCode = dataPoint.ParameterCode,
                Value = value,
                ProcessedValue = processedValue,
                DataType = dataPoint.DataType,
                Timestamp = DateTime.UtcNow,
                Quality = Quality.Good,
                Unit = dataPoint.Unit
            };
        }
        catch (Exception ex)
        {
            // 记录异常
            return new DeviceData
            {
                DataId = Guid.NewGuid().ToString(),
                PointId = dataPoint.PointId,
                DeviceId = DeviceInfo.DeviceId,
                ParameterCode = dataPoint.ParameterCode,
                Value = null,
                ProcessedValue = null,
                DataType = dataPoint.DataType,
                Timestamp = DateTime.UtcNow,
                Quality = Quality.Bad,
                Unit = dataPoint.Unit
            };
        }
    }
    
    // 其他方法实现...
    
    // 辅助方法：应用缩放和偏移
    private object ApplyScaleAndOffset(object value, double scale, double offset)
    {
        if (value == null)
            return null;
            
        switch (value)
        {
            case short s: return s * scale + offset;
            case ushort us: return us * scale + offset;
            case int i: return i * scale + offset;
            case uint ui: return ui * scale + offset;
            case long l: return l * scale + offset;
            case ulong ul: return ul * scale + offset;
            case float f: return f * scale + offset;
            case double d: return d * scale + offset;
            default: return value;
        }
    }
}
```

### 5.2 数据处理机制

1. **数据解析与转换**：
   - 将原始协议数据解析为统一的数据模型
   - 应用缩放因子和偏移量
   - 数据类型转换

2. **数据质量检查**：
   - 范围检查（最小值/最大值验证）
   - 一致性检查（与历史数据比较）
   - 有效性检查（数据格式验证）

3. **异常处理机制**：
   - 通信超时重试
   - 连接自动恢复
   - 数据质量标记

### 8.1 数据模型转换

```csharp
public class ProtocolConverter
{
    private readonly Dictionary<string, IDataModelConverter> _converters;
    
    public ProtocolConverter()
    {
        _converters = new Dictionary<string, IDataModelConverter>();
        // 注册内置转换器
        RegisterConverter(new ModbusToInternalConverter());
        RegisterConverter(new OpcUaToInternalConverter());
        RegisterConverter(new InternalToModbusConverter());
        RegisterConverter(new InternalToOpcUaConverter());
    }
    
    // 注册转换器
    public void RegisterConverter(IDataModelConverter converter)
    {
        var key = $"{converter.SourceProtocol}-{converter.TargetProtocol}";
        _converters[key] = converter;
    }
    
    // 转换数据模型
    public TTarget Convert<TSource, TTarget>(TSource sourceData, string sourceProtocol, string targetProtocol)
    {
        var key = $"{sourceProtocol}-{targetProtocol}";
        if (_converters.TryGetValue(key, out var converter))
        {
            return (TTarget)converter.Convert(sourceData);
        }
        throw new NotSupportedException($"Converter from {sourceProtocol} to {targetProtocol} is not supported");
    }
}
```
